              @mixin button-generator($width,$height,$background-color,$border,$border-radius,
                                      $box-shadow,$color,$font-size, $text-shadow) {
                width: $width;
                height: $height;
                line-height: $height;
                background-color: $background-color;
                border: $border;
                border-radius: $border-radius;
                box-shadow: $box-shadow;
                text-align: center;
                color: $color;
                font-size: $font-size;
                text-shadow: $text-shadow;  
              }
